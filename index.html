Tuiter

Simple Twitter API library for Nodejs

Intro

  var Tuiter = require('tuiter');

  var tuit = new Tuiter({
    "consumer_key" : "blablablablablabl"
    , "consumer_secret" : "blablablablab"
    , "access_token_key" : "blablablabla"
    , "access_token_secret" : "blablabla"
  });
					
  tuit.search({q: 'reptar'}, function(err, data){
    console.log(data);
  });

Installation

npm install tuiter
